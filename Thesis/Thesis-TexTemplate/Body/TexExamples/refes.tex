\section{引用格式的修改2}

\subsection*{增加~$\backslash$upcite~命令使显示的引用为上标形式}

方法1

\begin{verbatim}
\newcommand{\upcite}[1]{$^{\mbox{\scriptsize \cite{#1}}}$}
\end{verbatim}

方法2

\begin{verbatim}
\newcommand{\upcite}[1]{\textsuperscript{\textsuperscript{\cite{#1}}}}
\end{verbatim}

\section{引用示例}

一般引用 \verb"\cite{Bayes63:classical}" 产生如：\cite{Bayes63:classical}。

上标引用 \verb"\upcite{Yang_Hy200215}"，产生如：\upcite{Yang_Hy200215}

多个连续引用

\verb"\cite{Altman1997,Fujinoki2001,Hills,Joa1999,Yu_zhx2001,Zheng_wb200108}"，

产生如：\cite{Altman1997,Fujinoki2001,Hills,Joa1999,Yu_zhx2001,Zheng_wb200108}

\verb"\cite{Bayes63:classical,Hills,Joa1999,Yu_zhx2001,Zheng_wb200108}"，

产生如：\cite{Bayes63:classical,Hills,Joa1999,Yu_zhx2001,Zheng_wb200108}

多个上标连续引用

\verb"\upcite{Altman1997,Fujinoki2001,Hills,Joa1999,Yu_zhx2001,Zheng_wb200108}"，

产生如：\upcite{Altman1997,Fujinoki2001,Hills,Joa1999,Yu_zhx2001,Zheng_wb200108}
